```python
f = open('ege2026kp/26data/26-39.txt')
n, m = [int(x) for x in f.readline().split()]
a = sorted(int(x) for x in f)
b = []
for i in range(n):
    if 180 <= a[i] <= 200:
        b.append(a[i])
        a[i] = 0
a = [int(x) for x in a if x != 0]
m -= sum(b)
c = []
while sum(c) + a[0] <= m:
    c.append(a.pop(0))

c.sort(reverse=1)
for i in range(len(c)):
    for j in range(len(a)):
        if a[j] > c[i] and sum(c) - c[i] + a[j] <= m:
            c[i], a[j] = a[j], c[i]

print(len(b + c), sum(b + c))
```

```python
    road[x-h//2] += 1

st, end = -1, 0

    k0 = k
    k += road[i]

    if k0 == 0 and (k > 0 or i == l):

dark.sort()
lamp.sort()
s, over = 0, 0
for i in range(len(dark)):

        if lamp[j] >= dark[i]:
            s += lamp[j]
            over += lamp[j] - dark[i]
            lamp[j] = 0
            break
print(s, over)
```

# Одномерная динамика
### Вложенные коробки
```python
f = open('26.txt')
N = int(f.readline()) # кол-во коробок
a = [int(x) for x in f] # длины сторон коробок
a.sort(reverse=1) # сортируем по убыванию длины
max_zep = [0]*N # массив цепочек максимальной длины, оканчивающихся на этой коробке

for i in range(N):
    box = a[i]
    for j in range(i):
        prev_box = a[j]
        # Если коробка подходит по размерам, добавлем её в цепочку
        if prev_box - box >= 9:
            max_zep[i] = max(max_zep[i], max_zep[j]+1)
    
    if max_zep[i] == 0: max_zep[i] = 1
    
print(max(max_zep), max(a[i] for i in range(N) if max_zep[i] == max(max_zep)))
```

### Минимальная стоимость путешествия в зоны
```python
f = open('ege2026kp/26data/26-168.txt')
n, k = [int(x) for x in f.readline().split()]
trip = []
for s in f:
    frm, to, cost = [int(x) for x in s.split()]
    trip.append([frm, to, cost])
trip.sort()

min_cost = [10**20]*1000000 # мин. стоимость трансфера в зоне i
min_cost[1] = 0

for i in range(n):
    frm, to, cost = trip[i]
    sum_cost = min_cost[frm] + cost
    if sum_cost <= k:
        min_cost[to] = min(min_cost[to], sum_cost)

max_zone = max(i for i in range(1000000) if min_cost[i] != 10**20)
print(max_zone, k - min_cost[max_zone])
```
### Максимальное время занятости зала 
```python
f = open('ege2026kp/26data/26-160.txt')
n = int(f.readline())
events = []
for s in f:
    st, end, cost = [int(x) for x in s.split()]
    events.append([st, end, cost])
events.sort(key=lambda x:x[1]) # сортируем по времени окончания

max_dlit = [(0, 0)]*n # Макс. длительность и сумма аренды

for i in range(n):
    pred_dlit = [max_dlit[j] for j in range(i) if events[i][0] > events[j][1]]
    if pred_dlit:
        s, p = max(pred_dlit)
        max_dlit[i] = s + events[i][1] - events[i][0] + 1, p + events[i][2]
    else:
        max_dlit[i] = events[i][1] - events[i][0] + 1, events[i][2]
mx = max(max_dlit)
print(mx)
```
# Двумерная динамика
### Вложенные коробки
```python
f = open('ege2026kp/26data/26-165.txt')
N, K, Q = [int(x) for x in f.readline().split()]

boxes = []
for s in f:
    size, cost = [int(x) for x in s.split()]
    boxes.append([size, cost])
boxes.sort(reverse=1)

min_cost = [[10**20]*N for i in range(Q+1)]
# мин. стоимость упаковки из одной коробки равна стоимости этой коробки
min_cost[1] = [boxes[i][1] for i in range(N)] 

for q in range(2, Q+1):
    for i in range(N):
        # пред. цепочки коробок длины q-1
        pred_zep = [min_cost[q-1][j] for j in range(N) if boxes[j][0]-boxes[i][0]>=K]
        if pred_zep:
            min_cost[q][i] = min(pred_zep) + boxes[i][1]
min_c = min(min_cost[q])
print(min_c, min([boxes[i][0] for i in range(N) if min_cost[q][i] == min_c]))
```
